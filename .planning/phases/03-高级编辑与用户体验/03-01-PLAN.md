export function useDrawingTools() {
  const setPixel = useCanvasStore(state => state.setPixel);
  const deletePixel = useCanvasStore(state => state.deletePixel);
  const gridSize = useCanvasStore(state => state.gridSize);

  const applyBrush = useCallback((x: number, y: number, color: string, brushSize: number) => {
    const modified = new Set<string>();

    if (brushSize === 1) {
      const key = `${x},${y}`;
      setPixel(x, y, color);
      modified.add(key);
    } else {
      const radius = Math.floor(brushSize / 2);
      for (let dx = -radius; dx <= radius; dx++) {
        for (let dy = -radius; dy <= radius; dy++) {
          const px = x + dx;
          const py = y + dy;
          if (px >= 0 && px < gridSize.width && py >= 0 && py < gridSize.height) {
            const key = `${px},${py}`;
            if (!modified.has(key)) {
              setPixel(px, py, color);
              modified.add(key);
            }
          }
        }
      }
    }
  }, [setPixel, deletePixel, gridSize]);

  const applyEraser = useCallback((x: number, y: number, brushSize: number) => {
    const modified = new Set<string>();

    if (brushSize === 1) {
      const key = `${x},${y}`;
      deletePixel(x, y);
      modified.add(key);
    } else {
      const radius = Math.floor(brushSize / 2);
      for (let dx = -radius; dx <= radius; dx++) {
        for (let dy = -radius; dy <= radius; dy++) {
          const px = x + dx;
          const py = y + dy;
          if (px >= 0 && px < gridSize.width && py >= 0 && py < gridSize.height) {
            const key = `${px},${py}`;
            if (!modified.has(key)) {
              deletePixel(px, py);
              modified.add(key);
            }
          }
        }
      }
    }
  }, [deletePixel, gridSize]);

  // applyFill and applyColorPicker will be implemented in next tasks
  return { applyBrush, applyEraser };
}

Add applyFill flood fill implementation to useDrawingTools:

1. Extend useDrawingTools with applyFill function:
   - Signature: applyFill(startX: number, startY: number, fillColor: string)
   - Implement flood fill using BFS (Breadth-First Search) algorithm:
     * Get target color from start pixel (using canvasStore.grid.get(`${startX},${startY}`))
     * If target color equals fillColor, return early (no change needed)
     * Use Queue (array) for BFS traversal: start with [startX, startY]
     * Track visited pixels with Set<string> to avoid infinite loops
     * For each pixel in queue:
       - If pixel is target color: set to fillColor, add 4 neighbors (up, down, left, right) to queue
       - Skip visited pixels and out-of-bounds coordinates
     * Process pixels in chunks to avoid blocking JS thread for large fills (e.g., 100 pixels per frame using requestAnimationFrame)
   - Return count of modified pixels for progress tracking

Example implementation:
```typescript
const applyFill = useCallback(async (startX: number, startY: number, fillColor: string) => {
  const targetColor = grid.get(`${startX},${startY}`);
  if (targetColor === fillColor) return 0; // No change needed

  const visited = new Set<string>();
  const queue: [number, number][] = [[startX, startY]];
  const toProcess: [number, number][] = [];
  let modifiedCount = 0;

  const processChunk = () => {
    const chunk = toProcess.splice(0, 100); // Process 100 pixels per frame

    chunk.forEach(([x, y]) => {
      const key = `${x},${y}`;
      const pixelColor = grid.get(key);

      if (pixelColor === targetColor) {
        setPixel(x, y, fillColor);
        modifiedCount++;

        // Add neighbors
        const neighbors: [number, number][] = [
          [x + 1, y], [x - 1, y], [x, y + 1], [x, y - 1]
        ];

        neighbors.forEach(([nx, ny]) => {
          const nkey = `${nx},${ny}`;
          if (!visited.has(nkey) &&
              nx >= 0 && nx < gridSize.width &&
              ny >= 0 && ny < gridSize.height) {
            visited.add(nkey);
            toProcess.push([nx, ny]);
          }
        });
      }
    });

    if (toProcess.length > 0) {
      requestAnimationFrame(processChunk);
    }
  };

  visited.add(`${startX},${startY}`);
  processChunk();

  return modifiedCount;
}, [grid, setPixel, gridSize]);
```
